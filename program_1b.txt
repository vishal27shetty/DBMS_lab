-- Create the Restaurants table
CREATE TABLE Restaurants (
    id INT PRIMARY KEY,
    Name VARCHAR(255),
    Address VARCHAR(255),
    Cuisine VARCHAR(50),
    NearbyLandmarks VARCHAR(255),
    OnlineDelivery CHAR(3),
    FamousFor VARCHAR(255)
);
